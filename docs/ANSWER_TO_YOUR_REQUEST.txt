#pragma once
// SiliconForge — Scan Chain Insertion for DFT
// Converts regular DFFs to muxed-scan flip-flops and stitches scan chains.

#include "core/netlist.hpp"
#include <string>
#include <vector>

namespace sf {

struct ScanConfig {
    int max_chain_length = 100;           // Maximum FFs per scan chain
    std::string scan_in_prefix = "scan_in";      // Prefix for scan_in ports
    std::string scan_out_prefix = "scan_out";    // Prefix for scan_out ports
    std::string scan_enable_name = "scan_enable"; // Name of scan enable signal
};

struct ScanResult {
    int num_chains = 0;                   // Number of scan chains created
    int total_ffs = 0;                    // Total flip-flops in circuit
    std::vector<int> chain_lengths;       // Length of each chain
    std::string message;                  // Status/result message
};

class ScanInserter {
public:
    explicit ScanInserter(Netlist& nl) : nl_(nl) {}
    
    // Insert scan chains — modifies the netlist in place
    ScanResult insert(const ScanConfig& config = {});

private:
    Netlist& nl_;
};

} // namespace sf

DETAILED MEMBER DOCUMENTATION:

ScanConfig:
  int max_chain_length = 100
    - Default: 100 flip-flops per scan chain
    - Controls chain segmentation
    
  std::string scan_in_prefix = "scan_in"
    - Prefix for input port names
    - Generated names: "scan_in_0", "scan_in_1", etc.
    
  std::string scan_out_prefix = "scan_out"
    - Prefix for output port names
    - Generated names: "scan_out_0", "scan_out_1", etc.
    
  std::string scan_enable_name = "scan_enable"
    - Name of control signal that switches between normal/scan mode
    - Single signal for entire circuit

ScanResult:
  int num_chains = 0
    - Number of chains created (= total_ffs / max_chain_length, rounded up)
    
  int total_ffs = 0
    - Total flip-flops in original circuit
    
  std::vector<int> chain_lengths
    - One entry per chain with its length
    - Last chain may be shorter than max_chain_length
    
  std::string message
    - Human-readable status, e.g., "Inserted 2 scan chain(s) covering 150 FFs"

ScanInserter Class:
  Constructor:
    explicit ScanInserter(Netlist& nl)
      - Takes non-owning reference to netlist
      - Netlist must remain valid during insert() call
      - Does NOT copy netlist
  
  Public Method:
    ScanResult insert(const ScanConfig& config = {})
      - Main insertion algorithm
      - config: ScanConfig with parameters (uses defaults if not provided)
      - Returns: ScanResult with statistics
      - Side effects: Modifies netlist in-place
      
      Algorithm:
        1. Creates scan_enable net, marks as primary input
        2. Groups all flip-flops into chains of max_chain_length
        3. For each chain:
           a. Creates scan_in_N net (primary input)
           b. Inserts MUX gate before each DFF's D input
              - MUX selects between functional D and scan chain data
              - Select = scan_enable signal
           c. Chains FF Q outputs (previous FF Q → next FF MUX select)
           d. Creates scan_out_N from last FF's Q (primary output)
        4. Returns ScanResult with all statistics

EXAMPLE USAGE:

    Netlist nl;
    // ... build circuit with DFFs ...
    
    ScanConfig cfg;
    cfg.max_chain_length = 100;
    cfg.scan_enable_name = "scan_en";
    
    ScanInserter inserter(nl);
    ScanResult result = inserter.insert(cfg);
    
    std::cout << "Created " << result.num_chains << " chains" << std::endl;
    std::cout << "Total FFs: " << result.total_ffs << std::endl;
    std::cout << result.message << std::endl;

IMPLEMENTATION FILE: src/dft/scan_insert.cpp (71 lines)

Includes:

  const char* gate_type_str(GateType t)
    - Converts enum to string: "AND", "DFF", "MUX", etc.

Type Aliases:
  using NetId = int32_t;           (unique net identifier)
  using GateId = int32_t;          (unique gate identifier)

ENUM: GateType (18 types)
  
  I/O:        INPUT, OUTPUT
  Basic:      BUF, NOT
  Multi:      AND, OR, NAND, NOR
  2-input:    XOR, XNOR
  Mux:        MUX
  Sequential: DFF, DLATCH
  Special:    TRI, CONST0, CONST1
  Tristate:   BUFIF0, BUFIF1, NOTIF0, NOTIF1

STRUCT: Net (signal)
  
  NetId id                    - Unique identifier (index into nets_ vector)
  std::string name            - Signal name
  Logic4 value                - Current 4-state logic value
  Logic4 next_value           - Next value (after clock edge)
  std::vector<GateId> fanout  - Gates that read this net
  GateId driver = -1          - Gate that drives this net (-1 if primary input)

STRUCT: Gate (logic element)
  
  GateId id                   - Unique identifier (index into gates_ vector)
  GateType type               - Gate type
  std::string name            - Instance name
  std::vector<NetId> inputs   - Input net IDs (ORDER MATTERS for MUX)
  NetId output = -1           - Output net ID
  
  // For DFF/DLATCH only:
  NetId clk = -1              - Clock input net ID
  NetId reset = -1            - Reset input net ID (-1 if unused)
  Logic4 init_val = Logic4::ZERO  - Initial DFF value

  INPUT STRUCTURE FOR MUX:
    inputs[0] = select signal
    inputs[1] = data when select=1
    inputs[2] = data when select=0

  INPUT STRUCTURE FOR DFF:
    inputs[0] = D input
    (other fields in inputs unused)
    clk = clock net
    reset = reset net
    output = Q output

CLASS: Netlist (main data structure)

  CONSTRUCTION METHODS:
    
    NetId add_net(const std::string& name = "")
      - Creates new net (signal) in netlist
      - name: optional signal name; auto-generated if empty ("n0", "n1", ...)
      - Returns: unique NetId for this net
      - Side effects: appends to nets_ vector
    
    GateId add_gate(GateType type,
                   const std::vector<NetId>& inputs,
                   NetId output,
                   const std::string& name = "")
      - Creates combinational gate
      - type: gate type from GateType enum
      - inputs: vector of input net IDs (order matters!)
      - output: output net ID
      - name: optional gate name; auto-generated if empty
      - Returns: unique GateId
      - Side effects:
        * Appends to gates_ vector
        * Updates net.fanout for each input
        * Sets net.driver for output
    
    GateId add_dff(NetId d,
                  NetId clk,
                  NetId q,
                  NetId reset = -1,
                  const std::string& name = "")
      - Creates D flip-flop (sequential element)
      - d: D input net
      - clk: clock net
      - q: Q output net
      - reset: reset net (-1 if unused)
      - name: optional gate name
      - Returns: unique GateId
      - Side effects:
        * Appends to gates_ vector
        * Appends GateId to dffs_ vector
        * Updates fanout for d, clk, reset nets
        * Sets net.driver for q net

  I/O MARKING:
    
    void mark_input(NetId id)
      - Marks net as primary input
      - Appends id to pis_ vector
      - Typical usage: nl.mark_input(in_signal)
    
    void mark_output(NetId id)
      - Marks net as primary output
      - Appends id to pos_ vector
      - Typical usage: nl.mark_output(out_signal)

  ACCESS METHODS:
    
    Net& net(NetId id)
      - Mutable access to net by ID
      - Example: nl.net(5).value = Logic4::ONE
    
    const Net& net(NetId id) const
      - Const access to net
    
    Gate& gate(GateId id)
      - Mutable access to gate
      - Example: nl.gate(0).inputs[0] = new_net_id
    
    const Gate& gate(GateId id) const
      - Const access to gate

  QUERY METHODS:
    
    size_t num_nets() const
      - Total nets in circuit
    
    size_t num_gates() const
      - Total gates in circuit
    
    const std::vector<NetId>& primary_inputs() const
      - Vector of primary input net IDs (marked via mark_input)
      - Used to iterate circuit inputs
    
    const std::vector<NetId>& primary_outputs() const
      - Vector of primary output net IDs
    
    const std::vector<GateId>& flip_flops() const
      - Vector of DFF gate IDs (added via add_dff)
      - Critical for DFT tools: iterate with this!
    
    const std::vector<Net>& nets() const
      - All nets (const access)
    
    const std::vector<Gate>& gates() const
      - All gates (const access)

  ANALYSIS METHODS:
    
    std::vector<GateId> topo_order() const
      - Computes topological sort of gates
      - Uses Kahn's algorithm
      - Excludes DFF gates
      - Returns gates in evaluation order
      - Used by simulators, formal tools
    
    static Logic4 eval_gate(GateType type, const std::vector<Logic4>& inputs)
      - Pure static function
      - Evaluates single gate combinatorially
      - type: gate type
      - inputs: vector of Logic4 input values
      - Returns: Logic4 output value
      - Supports all combinational gates
      - Uses IEEE 1364-2005 truth tables
      
      Examples:
        eval_gate(GateType::AND, {ONE, ONE}) → ONE
        eval_gate(GateType::AND, {ONE, ZERO}) → ZERO
        eval_gate(GateType::OR, {ZERO, X}) → X
        eval_gate(GateType::MUX, {ONE, ZERO, ONE}) → ZERO

  MAINTENANCE:
    
    void print_stats() const
      - Prints statistics (net count, gate count, types, etc.)
    
    void clear()
      - Resets netlist to empty state

PRIVATE MEMBERS:
    
    std::vector<Net> nets_           - All nets, indexed by NetId
    std::vector<Gate> gates_         - All gates, indexed by GateId
    std::vector<NetId> pis_          - Primary input net IDs
    std::vector<NetId> pos_          - Primary output net IDs
    std::vector<GateId> dffs_        - DFF gate IDs

EXAMPLE USAGE:

    Netlist nl;
    
    // Create signals
    NetId in = nl.add_net("in");
    NetId clk = nl.add_net("clk");
    NetId q = nl.add_net("q");
    NetId out = nl.add_net("out");
    
    // Mark I/O
    nl.mark_input(in);
    nl.mark_input(clk);
    nl.mark_output(out);
    
    // Create DFF
    nl.add_dff(in, clk, q);
    
    // Create AND gate
    nl.add_gate(GateType::AND, {q, in}, out, "AND_1");
    
    // Query
    for (auto dff_id : nl.flip_flops()) {
        Gate& g = nl.gate(dff_id);
        std::cout << "DFF: " << g.name << std::endl;
    }
    
    // Evaluate
    Logic4 result = Netlist::eval_gate(GateType::AND, {ONE, ZERO});
    // result == ZERO

CRITICAL PATTERN - SAFE ITERATION:

    // DON'T: Direct iteration of flip_flops() while modifying
    for (auto& gid : nl.flip_flops()) {
        nl.add_gate(...);  // Modifies gates_ vector!
        Gate& g = nl.gate(gid);  // DANGLING REFERENCE if vector grew!
    }
    
    // DO: Cache IDs first
    std::vector<GateId> ff_ids(nl.flip_flops().begin(), nl.flip_flops().end());
    for (GateId gid : ff_ids) {
        nl.add_gate(...);  // Safe
        Gate& g = nl.gate(gid);  // Safe
    }
